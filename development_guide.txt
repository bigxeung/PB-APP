import React, { createContext, useState, useEffect, useContext } from 'react';
import { saveToken, getToken, clearTokens } from '../utils/storage';
import { authAPI } from '../services/auth';
import { UserResponse } from '../types';

interface AuthContextType {
  user: UserResponse | null;
  isLoading: boolean;
  isAuthenticated: boolean;
  login: (accessToken: string, refreshToken: string) => Promise<void>;
  logout: () => Promise<void>;
}

const AuthContext = createContext<AuthContextType | undefined>(undefined);

export const AuthProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [user, setUser] = useState<UserResponse | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  // ì´ˆê¸° ë¡œë“œ ì‹œ í† í° í™•ì¸
  useEffect(() => {
    checkAuth();
  }, []);

  const checkAuth = async () => {
    try {
      const token = await getToken();
      if (token) {
        const userData = await authAPI.getCurrentUser();
        setUser(userData);
      }
    } catch (error) {
      await clearTokens();
    } finally {
      setIsLoading(false);
    }
  };

  const login = async (accessToken: string, refreshToken: string) => {
    await saveToken(accessToken, refreshToken);
    const userData = await authAPI.getCurrentUser();
    setUser(userData);
  };

  const logout = async () => {
    await clearTokens();
    setUser(null);
  };

  return (
    <AuthContext.Provider
      value={{
        user,
        isLoading,
        isAuthenticated: !!user,
        login,
        logout,
      }}
    >
      {children}
    </AuthContext.Provider>
  );
};

export const useAuth = () => {
  const context = useContext(AuthContext);
  if (!context) throw new Error('useAuth must be used within AuthProvider');
  return context;
};

import React, { useState, useEffect } from 'react';
import { View, FlatList, RefreshControl } from 'react-native';
import { modelsAPI } from '../../services/models';
import ModelCard from '../../components/models/ModelCard';
import { LoraModel } from '../../types';

export default function ModelListScreen({ navigation }) {
  const [models, setModels] = useState<LoraModel[]>([]);
  const [loading, setLoading] = useState(false);
  const [refreshing, setRefreshing] = useState(false);
  const [page, setPage] = useState(0);
  const [hasMore, setHasMore] = useState(true);

  useEffect(() => {
    loadModels();
  }, []);

  const loadModels = async (refresh = false) => {
    if (loading || (!hasMore && !refresh)) return;

    try {
      setLoading(true);
      const currentPage = refresh ? 0 : page;
      const response = await modelsAPI.getPopularModels(currentPage);

      setModels(refresh ? response.content : [...models, ...response.content]);
      setPage(currentPage + 1);
      setHasMore(currentPage < response.totalPages - 1);
    } catch (error) {
      console.error(error);
    } finally {
      setLoading(false);
      setRefreshing(false);
    }
  };

  const onRefresh = () => {
    setRefreshing(true);
    loadModels(true);
  };

  const handleModelPress = (modelId: number) => {
    navigation.navigate('ModelDetail', { modelId });
  };

  return (
    <FlatList
      data={models}
      keyExtractor={(item) => item.id.toString()}
      renderItem={({ item }) => (
        <ModelCard model={item} onPress={() => handleModelPress(item.id)} />
      )}
      onEndReached={() => loadModels()}
      onEndReachedThreshold={0.5}
      refreshControl={
        <RefreshControl refreshing={refreshing} onRefresh={onRefresh} />
      }
    />
  );
}

import AsyncStorage from '@react-native-async-storage/async-storage';

const ACCESS_TOKEN_KEY = 'accessToken';
const REFRESH_TOKEN_KEY = 'refreshToken';

export const saveToken = async (accessToken: string, refreshToken: string) => {
  await AsyncStorage.setItem(ACCESS_TOKEN_KEY, accessToken);
  await AsyncStorage.setItem(REFRESH_TOKEN_KEY, refreshToken);
};

export const getToken = async () => {
  return await AsyncStorage.getItem(ACCESS_TOKEN_KEY);
};

export const clearTokens = async () => {
  await AsyncStorage.removeItem(ACCESS_TOKEN_KEY);
  await AsyncStorage.removeItem(REFRESH_TOKEN_KEY);
};
